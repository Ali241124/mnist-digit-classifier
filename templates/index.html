📄 app.py (Flask) 
from flask import Flask, jsonify, request
import torch
import numpy as np
from PIL import Image
from torchvision import transforms
import base64, io

app = Flask(__name__)
model = MNISTConvNet()
model.load_state_dict(torch.load('model/mnist_cnn.pth'))
model.eval()

@app.route('/predict', methods=['POST'])
def predict():
    data  = request.get_json()
    b64   = data['image'].split(',')[1]
    img   = Image.open(io.BytesIO(
                base64.b64decode(b64)
            )).convert('L').resize((28,28))
    
    tensor = transforms.ToTensor()(img).unsqueeze(0)
    probs = torch.softmax(model(tensor), dim=1)[0].detach().numpy()
    
    return jsonify({
        'prediction': int(np.argmax(probs)),
        'confidence': float(np.max(probs)),
        'all_probs' : probs.tolist()
    })
🖥️ cURL / Python 
# Python requests example
import requests, base64

with open("digit.png", "rb") as f:
    b64 = base64.b64encode(f.read()).decode()

resp = requests.post(
    "http://localhost:5000/predict",
    json={"image": f"data:image/png;base64,{b64}"}
)
result = resp.json()
print(f"Predicted: {result['prediction']}")
print(f"Confidence: {result['confidence']*100:.1f}%")

# Health check
requests.get("http://localhost:5000/health").json()
